// server.js // Express backend mínimo para comunicación con OpenAI y plantilla para Gemini. // npm i express node-fetch dotenv body-parser cors import express from "express"; import fetch from "node-fetch"; import dotenv from "dotenv"; import cors from "cors"; import bodyParser from "body-parser"; dotenv.config(); const app = express(); app.use(cors()); app.use(bodyParser.json()); const OPENAI_API_KEY = process.env.OPENAI_API_KEY; const OPENAI_MODEL = process.env.OPENAI_MODEL || "gpt-4o-mini"; // ajusta según tu cuenta // Opcional: GEMINI (plantilla). Debes completar GEMINI_API_KEY y endpoint si quieres usarlo. const GEMINI_API_KEY = process.env.GEMINI_API_KEY; const GEMINI_ENDPOINT = process.env.GEMINI_ENDPOINT; // p.ej. URL de PaLM/vertex (según Google) /** * /detect-language * Recibe { text } y devuelve { language: 'es' } (código ISO) usando OpenAI para detectar si se necesita. */ app.post("/detect-language", async (req, res) => { const { text } = req.body; if (!text) return res.status(400).json({ error: "text required" }); try { // Usamos OpenAI para detectar idioma de manera fiable const prompt = `Detecta el idioma del siguiente texto y responde solo con su código ISO-639-1 (p.ej. "es", "en", "fr"): "${text}"`; const response = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { Authorization: `Bearer ${OPENAI_API_KEY}`, "Content-Type": "application/json", }, body: JSON.stringify({ model: OPENAI_MODEL, messages: [{ role: "system", content: "Eres un detector de idioma." }, { role: "user", content: prompt }], temperature: 0, max_tokens: 5, }), }); const j = await response.json(); const out = j?.choices?.[0]?.message?.content?.trim()?.replace(/[".]/g, ""); const lang = out || (req.headers["accept-language"] ? req.headers["accept-language"].slice(0,2) : "es"); return res.json({ language: lang }); } catch (err) { console.error("detect-language err", err); return res.status(500).json({ error: "error detecting language" }); } }); /** * /assistant * Recibe: * { * service: "Risoterapia y Bienestar Natural", * userInput: "hola", * language: "es", * sessionId: "abc123", // opcional * systemPrompt: "..." // tu texto extenso de entrenamiento/personalidad * } * * Devuelve respuesta generada por OpenAI (y opcionalmente Gemini). */ app.post("/assistant", async (req, res) => { const { service, userInput, language = "es", sessionId = null, systemPrompt = "" } = req.body; if (!userInput) return res.status(400).json({ error: "userInput required" }); // Construimos prompt para el asistente: incluimos systemPrompt + datos del servicio y requerimientos const assistantSystem = ` Eres "Asistente May Roga", educador de salud y risoterapia. Responde siempre con amabilidad, respeto y calma. Cliente idioma: ${language} Servicio activo: ${service} Instrucciones clave: - Detecta y responde en el idioma indicado. - Si el contenido incluye un ejercicio con temporizador (ej: "respira por 1 minuto"), devuelve en el campo "actions" un array con el temporizador solicitado (en segundos). - Sigue la personalidad y la retórica de cuidado que se te proporcionó. SystemPrompt (entrenamiento del asistente): ${systemPrompt} `; const userMsg = `Usuario dice (en ${language}): "${userInput}". Responde de forma amable, breve y explica si hay ejercicios, indicando tiempos en segundos cuando corresponda.`; try { // Llamada a OpenAI Chat Completions const openaiResp = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { Authorization: `Bearer ${OPENAI_API_KEY}`, "Content-Type": "application/json", }, body: JSON.stringify({ model: OPENAI_MODEL, messages: [ { role: "system", content: assistantSystem }, { role: "user", content: userMsg }, ], temperature: 0.7, max_tokens: 600, }), }); const openaiJson = await openaiResp.json(); const openaiText = openaiJson?.choices?.[0]?.message?.content || ""; // Intentamos extraer instrucciones de temporizador desde el texto (heurística simple) // Buscamos patrones como "respira por X minuto(s)" o "X segundos" const timerMatches = []; const regexMin = /(?:respira|respiración|inhalar|exhalar).{0,30}?(\d+)\s*min(?:uto|utos)?/i; const regexSec = /(?:respira|respiración|inhalar|exhalar).{0,30}?(\d+)\s*seg(?:undo|undos)?/i; let m = openaiText.match(regexMin); if (m) timerMatches.push({ action: "silence_for", seconds: parseInt(m[1], 10) * 60, note: "ejercicio respiración (minutos detectado)" }); m = openaiText.match(regexSec); if (m) timerMatches.push({ action: "silence_for", seconds: parseInt(m[1], 10), note: "ejercicio respiración (segundos detectado)" }); // Respuesta final return res.json({ provider: "openai", text: openaiText, timers: timerMatches, raw: openaiJson, }); } catch (err) { console.error("assistant err", err); return res.status(500).json({ error: "assistant error" }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log("Server listening on", PORT));
